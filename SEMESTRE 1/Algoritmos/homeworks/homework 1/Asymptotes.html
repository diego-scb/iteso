import math
import numpy as np
import matplotlib.pyplot as plt

n = np.arange(0.1, 7., .1)
lg2 = [math.log2(x) for x in n]
nlg2 = [x*math.log2(x) for x in n]
labels=["logarithmic","linear","quasilinear","quadratic","cubic","exponential"]

def graficar():
    plt.figure(figsize=(10,6))
    plt.plot(n,lg2,'g-', label=labels[0] )
    plt.plot(n, n, 'b--', label=labels[1])
    plt.plot(n, nlg2, 'c-', label=labels[2])
    plt.plot(n, n**2, 'k-', label=labels[3])
    plt.plot(n, n**3, 'y-', label=labels[4])
    plt.plot(n, 2**n, 'r-', label=labels[5])
    plt.legend()
    plt.show()

graficar()
